Schema: family_schema



SCHEMA family_schema;


ENTITY Female
  SUBTYPE OF (Person);
END_ENTITY;

ENTITY Male
  SUBTYPE OF (Person);
END_ENTITY;

ENTITY Person
  ABSTRACT SUPERTYPE OF (ONEOF (Male,
                                Female));
  name : STRING;
  mother : OPTIONAL Female;
  father : OPTIONAL Male;
END_ENTITY;

END_SCHEMA;  -- family_schema
